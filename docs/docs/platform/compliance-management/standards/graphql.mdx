### Query Standards

```graphql
# Get all available standards
query GetStandards {
  standards {
    edges {
      node {
        id
        name
        shortName
        framework
        governingBody
        version
        domains
        status
        isPublic
        freeToUse
        description
        createdAt
        updatedAt
      }
    }
  }
}
```

```graphql
# Get a specific standard with its controls
query GetStandardWithControls($id: ID!) {
  standard(id: $id) {
    id
    name
    shortName
    framework
    governingBody
    domains
    description
    controls {
      edges {
        node {
          id
          displayID
          description
          category
          subcategory
          controlType
          status
        }
      }
    }
  }
}
```

### Create Standard

```graphql
mutation CreateStandard($input: CreateStandardInput!) {
  createStandard(input: $input) {
    standard {
      id
      name
      shortName
      framework
      governingBody
      version
      domains
      status
      isPublic
      freeToUse
    }
  }
}
```

**Input Example:**
```json
{
  "input": {
    "name": "Service Organization Control 2",
    "shortName": "SOC 2",
    "framework": "SOC2-2017",
    "governingBody": "AICPA",
    "version": "2017",
    "domains": ["confidentiality", "availability", "security"],
    "status": "ACTIVE",
    "isPublic": true,
    "freeToUse": false,
    "description": "SOC 2 defines criteria for managing customer data based on five trust service principles"
  }
}
```
